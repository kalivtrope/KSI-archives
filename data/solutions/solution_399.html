Jednoduché dotazy

Ako riešenie som sa rozhodol zverejniť len riešenie jednoduchých dotazov a databázu k nim potrebnú, na ktorých môžete pochopiť základy. Vypisovanie do súboru som tu neriešil, kedže verím, že riešiteľ KSI by si s tým poradil. Takto je to praktickejšie, ak sa len chcete pohrať s databázou.

from sqlalchemy import create_engine, Column, Integer, String, ForeignKey, ForeignKeyConstraint
from sqlalchemy.types import TIMESTAMP, DATE, TIME
from sqlalchemy.orm import sessionmaker, relationship
from sqlalchemy.ext.declarative import declarative_base
from datetime import datetime, time, date


# databse class section
Base = declarative_base()
class Penguin(Base):
     __tablename__ = 'Penguins'

     penguinID = Column(Integer, primary_key=True)
     firstName = Column(String)
     lastName = Column(String)
     birthday = Column(DATE)
     district = Column(Integer)
     vaccineNumber = Column(Integer)
     penguinPriority = Column(Integer)

     validcenters = relationship("ValidCenter", backref="Penguins")

     def __repr__(self):
        return "<Penguin(penguinID='%s', firstName='%s', lastName='%s')>" % (
                             self.penguinID, self.firstName, self.lastName)


class VaccinationCenter(Base):
     __tablename__ = 'VaccinationCenters'

     centerID = Column(Integer, primary_key=True)
     district = Column(Integer)
     workFrom = Column(TIME)
     workTill = Column(TIME)
     freeVaccines = Column(Integer)

     validToUser = relationship("ValidCenter", backref="VaccinationCenters")

     def __repr__(self):
        return "<Penguin(centerID='%s', freeVaccines='%s')>" % (
                             self.centerID, self.freeVaccines)


class WaitingRecord(Base):
     __tablename__ = 'WaitingList'

     registrationID = Column(Integer, primary_key=True)
     penguinID = Column(Integer, ForeignKey("Penguins.penguinID"))


     def __repr__(self):
        return "<Penguin(registrationID='%s', penguinID='%s')>" % (
                             self.registrationID, self.penguinID)


class ValidCenter(Base):
     __tablename__ = 'ValidCenters'

     penguinID = Column(Integer, ForeignKey("Penguins.penguinID"), primary_key=True)
     centerID = Column(Integer, ForeignKey("VaccinationCenters.centerID"), primary_key=True)

     def __repr__(self):
        return "<Penguin(penguinID='%s', centerID='%s')>" % (
                             self.penguinID, self.centerID)


class ValidTime(Base):
     __tablename__ = 'ValidTimes'

     penguinID = Column(Integer, ForeignKey("Penguins.penguinID"), primary_key=True)
     day = Column(Integer, primary_key = True)
     From = Column(TIME)
     To = Column(TIME)


     def __repr__(self):
        return "<Penguin(penguinID='%s', day='%s', From='%s', To='%s')>" % (
                             self.penguinID, self.day, self.From, self.To)

# functions working with the database

def createPenguins(comArgs, vaccinationsLimit, currentDate, oldAge):
    newPengs = [] 
    while comArgs != []:
        peng, comArgs = comArgs[0:7], comArgs[7:]
        bday = date.fromisoformat(peng[3])
        age = (currentDate - bday).days // 365
        if age <= 30 or bday > currentDate:
            basicPrio = 1 # ľahko zistíte ak si ich napíšete ako kombinácie do tabulky
            if peng[6] == '1':
                basicPrio += 1
            if age > oldAge:
                basicPrio += 2
            basicPrio += 4 * int(peng[5])
            #print(bday, age, basicPrio)
            newPengs.append(Penguin(penguinID = peng[0], firstName = peng[1], lastName = peng[2], birthday = bday,
                                    district = peng[4], vaccineNumber = peng[5], penguinPriority = basicPrio))
    return newPengs


def createCenters(comArgs):
    newCenters = [] 
    while comArgs != []:
        center, comArgs = comArgs[0:7], comArgs[7:]
        From = time(int(center[2]), int(center[3]))
        Till = time(int(center[4]), int(center[5]))
        newCenters.append(VaccinationCenter(centerID = center[0], district = center[1], workFrom = From, workTill = Till, freeVaccines = center[6]))
    return newCenters


def getRegNumber():
    registrations = 0
    while True:
        yield registrations
        registrations += 1


def findCentersInDistrict(session, dist):
    centers = []
    for center,_ in session.query(VaccinationCenter.centerID, VaccinationCenter.district).filter(VaccinationCenter.district == dist):
        centers.append(center)
    return centers


def allCenters(session):
    return set([center[0] for center in session.query(VaccinationCenter.centerID)])


def alwaysAvalible():
    return [[i, time(0,0), time(23,59)] for i in range(7)]


def addDaysToValidTimes(session, comArgs, i):
    avalibleTimes = []
    always = alwaysAvalible()
    if comArgs[i] == "ALWAYS":
        avalibleTimes = [ValidTime(penguinID = comArgs[0], day = a[0], From = a[1], To = a[2]) for a in always]
    else:
        while i < len(comArgs):
        # dayDefinition, short so it is not an abomination in lower code
            dD = []
            i += 1
            while len(comArgs) > i and comArgs[i].isdigit():
                dD.append(int(comArgs[i]))
                i += 1
            if len(dD) == 1:
                avalibleTimes.append(ValidTime(penguinID = comArgs[0], day = dD[0], From = always[dD[0]][1], To = always[dD[0]][2]))
            else:
                timeFrom = time(dD[1], dD[2])
                timeTo = time(dD[3], dD[4])
                avalibleTimes.append(ValidTime(penguinID = comArgs[0], day = dD[0], From = timeFrom, To = timeTo))
    session.add_all(avalibleTimes)


def registerForVaccination(comArgs, session, vaccinationsLimit, regID):
    penguin = session.query(Penguin.district, Penguin.vaccineNumber).filter(Penguin.penguinID == comArgs[0]).all()
    if len(penguin) == 0:
        return None
    else:
        dist, vacNum = penguin[0]
    if vacNum >= vaccinationsLimit:
        return None
    # update waitingList
    session.add(WaitingRecord(registrationID = next(regID), penguinID = comArgs[0]))
    # move on to validCenters
    centers = []
    i = 2
    if comArgs[1] == "CENTERS":
        all_centers = allCenters(session)
        while len(comArgs) > i and comArgs[i].isdigit():
            if int(comArgs[i]) in all_centers:
                centers.append(ValidCenter(penguinID = comArgs[0], centerID = comArgs[i]))
            i+=1
    if comArgs[1] == "ALL" or centers == []:
        districtCenters = findCentersInDistrict(session, dist)
        for center in districtCenters:
            centers.append(ValidCenter(penguinID = comArgs[0], centerID = center))
    session.add_all(centers)
    # finally times
    addDaysToValidTimes(session, comArgs, i)


def changeValidCenters(comArgs, session):
    all_centers = allCenters(session)

    for center in comArgs[1:]:
        if center[0] == '-':
            session.query(ValidCenter).filter(ValidCenter.centerID == int(center[1:]), ValidCenter.penguinID == comArgs[0]).delete(synchronize_session='evaluate')
        else:
            session.add(ValidCenter(penguinID = comArgs[0], centerID = center))
    if session.query(ValidCenter).filter(ValidCenter.penguinID == comArgs[0]).all() == []:
        session.add(ValidCenter(penguinID = comArgs[0], centerID = comArgs[-1][1:]))


def changeValidTimes(comArgs, session):
    # get rid of days that we want to remove
    toDelete = []
    for i in range(len(comArgs)-1, 0, -1):
        if comArgs[i] == "NOT":
            toDelete.append(i)
            session.query(ValidTime).filter(ValidTime.day == int(comArgs[i-1]), ValidTime.penguinID == comArgs[0]).delete(synchronize_session='evaluate')
    for i in toDelete:
        comArgs = comArgs[0:i - 2] + comArgs[i+1:]
    # add days like before
    if len(comArgs) > 1:
        addDaysToValidTimes(session, comArgs, 1)

def printQuerryRes(result):
    for row in result:
        print(("|").join([str(column) for column in row]))

def printValidTimes(session, comArgs):
    result = []
    i = 0
    while i < len(comArgs):
        if comArgs[i] == "ID" and i + 2 < len(comArgs) and comArgs[i+2] == "DAY":
            result += (session.query(ValidTime.penguinID, ValidTime.day, ValidTime.From, ValidTime.To).\
                          filter(ValidTime.penguinID == comArgs[i+1] , ValidTime.day == comArgs[i+3]).all())
            i += 4
        elif comArgs[i] == "ID":
            result += (session.query(ValidTime.penguinID, ValidTime.day, ValidTime.From, ValidTime.To).\
                          filter(ValidTime.penguinID == comArgs[i+1]).all())
            i += 2
        elif comArgs[i] == "DAY" and i + 2 < len(comArgs) and comArgs[i+2].isdigit():
            result += (session.query(ValidTime.penguinID, ValidTime.day, ValidTime.From, ValidTime.To).\
                          filter(ValidTime.day == comArgs[i+1], ValidTime.From <= time(int(comArgs[i+2]), int(comArgs[i+3])), ValidTime.To >= time(int(comArgs[i+4]), int(comArgs[i+5]))).all())
            i += 6
        else:
            result +=(session.query(ValidTime.penguinID, ValidTime.day, ValidTime.From, ValidTime.To).\
                          filter(ValidTime.day == comArgs[i+1]).all())
            i += 2
    finalResult = []
    for possibleRes in result:
        if possibleRes not in finalResult:
            finalResult.append(possibleRes)

    printQuerryRes(sorted(finalResult))

# Coprogram that works like a query for our commands
def databaseOperator(vaccinationsLimit, currentDate, oldAge, printFile):
    # setup actual database here
    engine = create_engine('sqlite:///:memory:', echo=False)
    Base.metadata.create_all(engine)
    Session = sessionmaker(bind=engine)
    session = Session()
    # initialize our regID generator
    regID = getRegNumber()
    # do not change this
    while True:
        newCommand = yield
        # add your logic here
        command, comArgs = newCommand.split(" ", 1)
        comArgs = comArgs.split(" ")
        #print(command, comArgs)
        if command == "CREATEPENGUIN":
            newPenguins = createPenguins(comArgs, vaccinationsLimit, currentDate, oldAge)
            session.add_all(newPenguins)

        elif command == "CREATECENTER":
            newVacCenters = createCenters(comArgs)
            session.add_all(newVacCenters)

        elif command == "REGISTERPENGUIN":
            registerForVaccination(comArgs, session, vaccinationsLimit, regID)

        elif command == "CHANGEREGISTRATIONCENTERS":
            changeValidCenters(comArgs, session)

        elif command == "CHANGEREGISTRATIONTIMES":
            changeValidTimes(comArgs, session)

        elif command == "PRINTREGISTERED":
            waiting = session.query(WaitingRecord.registrationID, Penguin.penguinID, Penguin.firstName, Penguin.lastName, Penguin.penguinPriority).\
                      filter(Penguin.penguinID == WaitingRecord.penguinID).order_by(WaitingRecord.registrationID).all()[0:int(comArgs[0])]
            printQuerryRes(waiting)

        elif command == "PRINTFREECENTERS":
            freeCenters = session.query(VaccinationCenter.centerID, VaccinationCenter.freeVaccines).\
                          filter(VaccinationCenter.freeVaccines > 0, VaccinationCenter.district == int(comArgs[0])).order_by(VaccinationCenter.centerID).all()
            printQuerryRes(freeCenters)

        elif command == "PRINTVALIDTIMES":
            printValidTimes(session, comArgs)

        session.commit()
