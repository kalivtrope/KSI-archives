Řešení

Řešenie môže byť napríklad následujúca funkcia.

def count_flips(bs):
    # Stav 0 vyjadruje, ze nemame nacitany ziadny znak z podretazca "010".
    state = 0
    # Counter pocita pocet potrebnych zmien.
    counter = 0
    for num in range(len(bs)):
        # Ak sme v stave 0 a nacitali sme nulu, tak sa prepneme do
        # stavu 1, kde budeme cakat na znak 1.
        if state == 0 and bs[num] == '0':
            state = 1
        # Po nacitani znaku 1 sa prepneme do stavu 2.
        elif state == 1 and bs[num] == '1':
            state = 2
        # Ak sme v stave 2 nacitali 0, tak to znamena,
        # ze sme uz nacitali cely retazec "010".
        # Tedy musime v tomto retazci aspon jeden znak zmenit.
        # Zmenime posledny znak, aby sme pripadne pokazili aj
        # nasledujuci podretazec "010", ktory by mohol zacinat
        # v poslednej nule predchadzajuceho vyskytu "010".
        elif state == 2 and bs[num] == '0':
            state = 0
            counter += 1
        # Ak sme v stave 2 a namiesto 0 sme nacitali 1,
        # znamena to, ze mozeme zacat kontrolovat opat
        # od stavu 0 (lebo sme nacitali 2 jednicky za sebou).
        elif state == 2 and bs[num] == '1':
            state = 0
    return counter
